6   package gov.nist.secauto.metaschema.core.metapath.function.library;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
11  import gov.nist.secauto.metaschema.core.metapath.function.IFunction;
12  import gov.nist.secauto.metaschema.core.metapath.item.IItem;
13  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IBooleanItem;
14  
15  import java.util.List;
16  
17  import edu.umd.cs.findbugs.annotations.NonNull;
18  
19  public final class FnTrue {
20    @NonNull
21    static final IFunction SIGNATURE = IFunction.builder()
22        .name("true")
23        .namespace(MetapathConstants.NS_METAPATH_FUNCTIONS)
24        .deterministic()
25        .contextIndependent()
26        .focusIndependent()
27        .returnType(IBooleanItem.class)
28        .returnOne()
29        .functionHandler(FnTrue::execute)
30        .build();
31  
32    private FnTrue() {
33      // disable construction
34    }
35  
36    @SuppressWarnings("unused")
37    @NonNull
38    private static ISequence<IBooleanItem> execute(@NonNull IFunction function,
39        @NonNull List<ISequence<?>> arguments,
40        @NonNull DynamicContext dynamicContext,
41        IItem focus) {
42  
43      return ISequence.of(IBooleanItem.TRUE);
44    }
45  }

50  public class ValidateContentUsingModuleCommand
51      extends AbstractValidateContentCommand {
52    @NonNull
53    private static final String COMMAND = "validate-content";
54  
55    @Override
56    public String getName() {
57      return COMMAND;
58    }
59  
60    @Override
61    public String getDescription() {
62      return "Verify that the provided resource is well-formed and valid to the provided Module-based model.";
63    }
64  
65    @Override
66    public Collection<? extends Option> gatherOptions() {
67      Collection<? extends Option> orig = super.gatherOptions();
68  
69      List<Option> retval = new ArrayList<>(orig.size() + 1);
70      retval.addAll(orig);
71      retval.add(MetaschemaCommandSupport.METASCHEMA_OPTION);
72  
73      return CollectionUtil.unmodifiableCollection(retval);
74    }
75  
76    @Override
77    public ICommandExecutor newExecutor(CallingContext callingContext, CommandLine commandLine) {
78      return new OscalCommandExecutor(callingContext, commandLine);
79    }
80  
81    private final class OscalCommandExecutor
82        extends AbstractValidationCommandExecutor {
83  
84      private Path tempDir;
85      private IModule module;
86  
87      private OscalCommandExecutor(
88          @NonNull CallingContext callingContext,
89          @NonNull CommandLine commandLine) {
90        super(callingContext, commandLine);
91      }
92  
93      private Path getTempDir() throws IOException {
94        if (tempDir == null) {
95          tempDir = Files.createTempDirectory("validation-");
96          tempDir.toFile().deleteOnExit();
97        }
98        return tempDir;
99      }
100 
101     @NonNull
102     private IModule getModule(@NonNull Set<IConstraintSet> constraintSets)
103         throws MetaschemaException, IOException {
104       URI cwd = Paths.get("").toAbsolutePath().toUri();
105 
106       if (module == null) {
107         String moduleName
108             = ObjectUtils.requireNonNull(getCommandLine().getOptionValue(MetaschemaCommandSupport.METASCHEMA_OPTION));
109         URI moduleUri;
110 
111         try {
112           moduleUri = UriUtils.toUri(moduleName, cwd);
113         } catch (URISyntaxException ex) {
114           IOException newEx = new IOException( // NOPMD - intentional
115               String.format("Cannot load module as '%s' is not a valid file or URL.", moduleName));
116           newEx.addSuppressed(ex);
117           throw newEx;
118         }
119 
120         assert moduleUri != null;
121 
122         ExternalConstraintsModulePostProcessor postProcessor
123             = new ExternalConstraintsModulePostProcessor(constraintSets);
124 
125         ModuleLoader loader = new ModuleLoader(CollectionUtil.singletonList(postProcessor));
126 
127         // BindingModuleLoader loader
128         // = new BindingModuleLoader(new DefaultBindingContext(),
129         // CollectionUtil.singletonList(postProcessor));
130 
131         loader.allowEntityResolution();
132         module = loader.load(moduleUri);
133       }
134       assert module != null;
135       return module;
136     }
137 
138     @NonNull
139     private IModule getModule() {
140       // should be initialized already
141       return ObjectUtils.requireNonNull(module);
142     }
143 
144     @Override
145     protected IBindingContext getBindingContext(@NonNull Set<IConstraintSet> constraintSets)
146         throws MetaschemaException, IOException {
147 
148       IBindingContext retval = new DefaultBindingContext();
149       retval.registerModule(getModule(constraintSets), getTempDir());
150       return retval;
151     }
152 
153     @Override
154     public List<Source> getXmlSchemas(@NonNull URL targetResource) throws IOException {
155       Path schemaFile = Files.createTempFile(getTempDir(), "schema-", ".xml");
156       assert schemaFile != null;
157       IMutableConfiguration<SchemaGenerationFeature<?>> configuration = new DefaultConfiguration<>();
158       ISchemaGenerator.generateSchema(getModule(), schemaFile, SchemaFormat.XML, configuration);
159       return ObjectUtils.requireNonNull(List.of(
160           XmlUtil.getStreamSource(schemaFile.toUri().toURL())));
161     }
162 
163     @Override
164     public JSONObject getJsonSchema(@NonNull JSONObject json) throws IOException {
165       Path schemaFile = Files.createTempFile(getTempDir(), "schema-", ".json");
166       assert schemaFile != null;
167       IMutableConfiguration<SchemaGenerationFeature<?>> configuration = new DefaultConfiguration<>();
168       ISchemaGenerator.generateSchema(getModule(), schemaFile, SchemaFormat.JSON, configuration);
169       try (BufferedReader reader = ObjectUtils.notNull(Files.newBufferedReader(schemaFile, StandardCharsets.UTF_8))) {
170         return JsonUtil.toJsonObject(reader);
171       }
172     }
173   }

6   package gov.nist.secauto.metaschema.core.model;
7   
8   import java.util.Collection;
9   
10  import javax.xml.namespace.QName;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  import edu.umd.cs.findbugs.annotations.Nullable;
14  
15  public interface IModelDefinition extends IDefinition, IContainer {
16    @Override
17    default boolean hasChildren() {
18      return !getFlagInstances().isEmpty();
19    }
20  
21    /**
22     * Retrieves a flag instance, by the flag's effective name, that is defined on
23     * the containing definition.
24     *
25     * @param name
26     *          the flag's name
27     * @return the matching flag instance, or {@code null} if there is no flag
28     *         matching the specified name
29     */
30    @Nullable
31    IFlagInstance getFlagInstanceByName(@NonNull QName name);
32  
33    /**
34     * Retrieves the flag instances for all flags defined on the containing
35     * definition.
36     *
37     * @return the flags
38     */
39    @NonNull
40    Collection<? extends IFlagInstance> getFlagInstances();
41  
42    /**
43     * Retrieves the flag instance to use as as the property name for the containing
44     * object in JSON who's value will be the object containing the flag.
45     *
46     * @return the flag instance if a JSON key is configured, or {@code null}
47     *         otherwise
48     */
49    // TODO: remove once moved to the instance side
50    // TODO: Reconsider using toFlagName in favor of getReferencedDefinitionQName
51    @Nullable
52    IFlagInstance getJsonKey();
53  }

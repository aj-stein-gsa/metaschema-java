11  import edu.umd.cs.findbugs.annotations.Nullable;
12  import nl.talsmasoftware.lazy4j.Lazy;
13  
14  class AssemblyOrphanedDefinitionDataNodeItemImpl
15      extends AbstractOrphanedDefinitionNodeItem<IAssemblyDefinition, IAssemblyInstance>
16      implements IAssemblyNodeItem,
17      IFeatureModelContainerItem,
18      IFeatureRequiredDataItem,
19      IFeatureOrhpanedDefinitionModelNodeItem<IAssemblyDefinition, IAssemblyInstance> {
20    @NonNull
21    private final Lazy<ModelContainer> model;
22    @NonNull
23    private final Object value;
24  
25    public AssemblyOrphanedDefinitionDataNodeItemImpl(
26        @NonNull IAssemblyDefinition definition,
27        @Nullable URI baseUri,
28        @NonNull Object value,
29        @NonNull INodeItemGenerator generator) {
30      super(definition, baseUri);
31      this.value = value;
32      this.model = ObjectUtils.notNull(Lazy.lazy(generator.newDataModelSupplier(this)));
33    }
34  
35    @SuppressWarnings("null")
36    @Override
37    public ModelContainer getModel() {
38      return model.get();
39    }
40  
41    @Override
42    public Object getValue() {
43      return value;
44    }
45  }
